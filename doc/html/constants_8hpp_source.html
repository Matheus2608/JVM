35// Access Flags (podem ser combinadas com bitwise OR)
36enum class AccessFlags : u2 {
37 ACC_PUBLIC = 0x0001, // Declarada pública
38 ACC_PRIVATE = 0x0002, // Declarada privada
39 ACC_PROTECTED = 0x0004, // Declarada protegida
40 ACC_STATIC = 0x0008, // Static
41 ACC_FINAL = 0x0010, // Declarada final
42 ACC_SUPER = 0x0020, // Chama métodos de superclasse (classe)
43 ACC_SYNCHRONIZED= 0x0020, // Synchronized (método)
44 ACC_VOLATILE = 0x0040, // Volatile (campo)
45 ACC_BRIDGE = 0x0040, // Bridge method
46 ACC_TRANSIENT = 0x0080, // Transient (campo)
47 ACC_VARARGS = 0x0080, // Varargs
48 ACC_NATIVE = 0x0100, // Native
49 ACC_INTERFACE = 0x0200, // É interface
50 ACC_ABSTRACT = 0x0400, // Declarada abstrata
51 ACC_STRICT = 0x0800, // Strictfp
52 ACC_SYNTHETIC = 0x1000, // Não presente no fonte
53 ACC_ANNOTATION = 0x2000, // Tipo annotation
54 ACC_ENUM = 0x4000 // Tipo enum
55};
