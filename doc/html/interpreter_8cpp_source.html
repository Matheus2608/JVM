180void Interpreter::execute(const class_info& cls, const method_info& method) {
181 const code_attribute* code = findCode(method);
182 if (!code)
183 throw std::runtime_error("Interpretador: metodo nao tem atributo Code");
184
185 Frame f;
186 f.cls = &cls;
187 f.method = &method;
188 f.code = code;
189 f.pc = 0;
190 f.locals.resize(code->max_locals, Value::fromInt(0));
191
192 frame_stack_.push(std::move(f));
193 run();
194}
195
196void Interpreter::run() {
197 while (!halted_ && !frame_stack_.empty()) {
198 uint8_t opcode = fetchU1();
199
200 auto it = dispatch_table_.find(opcode);
201 if (it == dispatch_table_.end()) {
202 std::ostringstream oss;
203 oss << "Opcode nao implementado: 0x"
204 << std::hex << std::setw(2) << std::setfill('0')
205 << static_cast<int>(opcode);
206 throw std::runtime_error(oss.str());
207 }
208
209 it->second(); // executa o handler do opcode
210 }
211}

242void Interpreter::branch(int32_t offset) {
243 // offset é relativo ao início da instrução (pc já avançou 1 para o opcode
244 // e mais N para os operandos); ajustamos subtraindo o que foi consumido.
245 Frame& f = currentFrame();
246 f.pc += offset - 3; // 3 = 1 (opcode) + 2 (operandos de branch padrão)

249const code_attribute* Interpreter::findCode(const method_info& method) const {
250 for (const attribute_info& attr : method.attributes)

1188// tableswitch — salto indexado: se `index` estiver em [low, high], desvia
1189// para o offset da tabela correspondente; caso contrário, usa o default.
1190// Os offsets são relativos ao endereço do próprio opcode (`start`).
1191void Interpreter::op_tableswitch() {
1192 Frame& f = currentFrame();
1193 size_t start = f.pc - 1; // endereço do opcode (fetchU1() já avançou o pc)
1194 while (f.pc % 4 != 0) f.pc++; // pula bytes de padding até alinhamento de 4
1195
1196 int32_t def = fetchS4();
1197 int32_t low = fetchS4();
1198 int32_t high = fetchS4();
1199
1200 int32_t index = f.pop().data.i;
1201
1202 int32_t offset;
1203 if (index < low || index > high) {
1204 offset = def;
1205 } else {
1206 f.pc += static_cast<size_t>(index - low) * 4; // pula até a entrada certa
1207 offset = fetchS4();
1208 }
1209
1210 f.pc = start + static_cast<size_t>(offset);
1211}

1213// lookupswitch — salto por tabela de pares (match, offset); usa default se
1214// nenhum match for encontrado.
1215void Interpreter::op_lookupswitch() {
1216 Frame& f = currentFrame();
1217 size_t start = f.pc - 1;
1218 while (f.pc % 4 != 0) f.pc++;
1219
1220 int32_t def = fetchS4();
1221 int32_t npairs = fetchS4();
1222
1223 int32_t key = f.pop().data.i;
1224 int32_t offset = def;
1225 for (int32_t j = 0; j < npairs; j++) {
1226 int32_t match = fetchS4();
1227 int32_t off = fetchS4();
1228 if (match == key) {
1229 offset = off;
1230 break;
1231 }
1232 }
1233
1234 f.pc = start + static_cast<size_t>(offset);
1235}

1268// Helpers de invocação de métodos (locais a este arquivo)
1269//
1270// Cada argumento ocupa um único Value na pilha de operandos deste interpretador
1271// (long/double inclusive). Porém, nas variáveis locais, o javac assume que
1272// long/double consomem 2 slots — por isso o índice do slot avança conforme a
1273// "largura" de cada parâmetro, ainda que apenas um Value seja copiado.
1274// -----------------------------------------------------------------------------
1275namespace {
1276
1277// Largura em slots de cada parâmetro do descritor (long/double = 2, resto = 1).
1278std::vector<int> parseParamWidths(const std::string& descriptor) {
1279 std::vector<int> widths;
1280 size_t i = 1; // pula '('
1281 while (i < descriptor.size() && descriptor[i] != ')') {
1282 char c = descriptor[i];
1283 if (c == '[') { i++; continue; } // prefixo de array; a ref é category 1
1284 if (c == 'L') { // objeto: consome até ';'
1285 while (i < descriptor.size() && descriptor[i] != ';') i++;
1286 i++; // pula ';'
1287 widths.push_back(1);
1288 continue;
1289 }
1290 widths.push_back((c == 'J' || c == 'D') ? 2 : 1); // long/double = category 2
1291 i++;
1292 }
1293 return widths;
1294}

1298 for (const attribute_info& attr : method.attributes)
1299 if (attr.code_data)
1300 return attr.code_data.get();

1304// Resolve um método percorrendo a classe e suas superclasses. Em caso de
1305// sucesso, *decl recebe a classe onde o método foi efetivamente encontrado.
1306const method_info* resolveMethod(ClassLoader& loader, const class_info& start,
1307 const std::string& name, const std::string& descriptor,
1308 const class_info** decl) {
1309 const class_info* cls = &start;
1310 while (true) {
1311 if (const method_info* m = findMethod(*cls, name, descriptor)) {
1312 if (decl) *decl = cls;
1313 return m;
1314 }
1315 if (cls->super_class == 0)
1316 return nullptr;
1317 std::string super = classNameFromConstantPool(*cls, cls->super_class);
1318 try {
1319 cls = &loader.load(super);
1320 } catch (...) {
1321 return nullptr;
1322 }
1323 }
1324}

1326// Monta o frame do método chamado a partir dos argumentos já isolados e o
1327// empilha. `self` é Value::null() para chamadas estáticas.
1328void pushCalleeFrame(FrameStack& fs, const class_info& decl, const method_info& method,
1329 const code_attribute* code, const std::vector<Value>& args,
1330 const std::vector<int>& widths, bool hasThis, Value self) {
1331 Frame nf;
1332 nf.cls = &decl;
1333 nf.method = &method;
1334 nf.code = code;
1335 nf.pc = 0;
1336 nf.locals.assign(code->max_locals, Value::fromInt(0));
1337
1338 int slot = 0;
1339 if (hasThis)
1340 nf.locals[slot++] = self;
1341 for (size_t k = 0; k < args.size(); k++) {
1342 nf.locals[slot] = args[k];
1343 slot += widths[k];
1344 }
1345 fs.push(std::move(nf));
1346}

1348// Invocação estática: sem receiver. Resolve o método e empilha o frame.
1349void invokeStaticMethod(ClassLoader& loader, FrameStack& fs,
1350 const std::string& class_name, const std::string& name,
1351 const std::string& descriptor) {
1352 Frame& caller = fs.top();
1353 std::vector<int> widths = parseParamWidths(descriptor);
1354 std::vector<Value> args(widths.size());
1355 for (size_t k = widths.size(); k-- > 0; )
1356 args[k] = caller.pop();
1357
1358 const class_info* target = nullptr;
1359 try { target = &loader.load(class_name); } catch (...) { target = nullptr; }
1360
1361 const class_info* decl = nullptr;
1362 const method_info* m = target ? resolveMethod(loader, *target, name, descriptor, &decl) : nullptr;
1363 const code_attribute* code = m ? findCodeOf(*m) : nullptr;
1364 if (!m || !code)
1365 throw std::runtime_error("invokestatic: metodo nao encontrado ou sem Code: " +
1366 class_name + "." + name + descriptor);
1367
1368 pushCalleeFrame(fs, *decl, *m, code, args, widths, false, Value::null());
1369}

1371// Invocação de instância (special/virtual/interface). Em virtualDispatch a
1372// resolução parte da classe real do receiver no heap; caso contrário, da classe
1373// estática do constant pool. Métodos sem corpo (ex.: java/lang/Object.<init> ou
1374// nativos) viram no-op — os argumentos e o receiver já foram desempilhados.
1375void invokeInstanceMethod(ClassLoader& loader, FrameStack& fs, Heap& heap,
1376 const std::string& static_class, const std::string& name,
1377 const std::string& descriptor, bool virtualDispatch) {
1378 Frame& caller = fs.top();
1379 std::vector<int> widths = parseParamWidths(descriptor);
1380 std::vector<Value> args(widths.size());
1381 for (size_t k = widths.size(); k-- > 0; )
1382 args[k] = caller.pop();
1383 Value self = caller.pop();
1384
1385 if (heap.isNull(self.data.ref))
1386 throw std::runtime_error("NullPointerException");
1387
1388 std::string start_class = static_class;
1389 if (virtualDispatch) {
1390 // Despacho dinâmico: usa a classe concreta do objeto referenciado.
1391 try { start_class = heap.object(self.data.ref).class_name; } catch (...) {}
1392 }
1393
1394 const class_info* target = nullptr;
1395 try { target = &loader.load(start_class); } catch (...) { target = nullptr; }
1396
1397 const class_info* decl = nullptr;
1398 const method_info* m = target ? resolveMethod(loader, *target, name, descriptor, &decl) : nullptr;
1399 const code_attribute* code = m ? findCodeOf(*m) : nullptr;
1400 if (!m || !code)
1401 return; // no-op: construtor de Object, método nativo, etc.
1402
1403 pushCalleeFrame(fs, *decl, *m, code, args, widths, true, self);
1404}

1817// Helper local: percorre a cadeia de superclasses verificando se `sub` é
1818// subtipo de `target`. Usado por checkcast e instanceof. Tudo que herda de
1819// java/lang/Object é subtipo dela mesmo quando a stdlib não está no classpath.
1820namespace {
1821bool isSubtypeOf(ClassLoader& loader, const std::string& sub, const std::string& target) {
1822 if (sub == target || target == "java/lang/Object")
1823 return true;
1824
1825 std::string current = sub;
1826 while (true) {
1827 const class_info* cls = nullptr;
1828 try {
1829 cls = &loader.load(current);
1830 } catch (...) {
1831 return false; // classe não encontrada no classpath
1832 }
1833
1834 if (cls->super_class == 0)
1835 return false; // chegamos em java/lang/Object sem casar
1836
1837 std::string super = classNameFromConstantPool(*cls, cls->super_class);
1838 if (super == target)
1839 return true;
1840 if (super == "java/lang/Object")
1841 return false;
1842 current = super;
1843 }
1844}
