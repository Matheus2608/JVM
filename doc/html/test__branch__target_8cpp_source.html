18// Captura stdout durante a execução de disassembleCode.
19static std::string capture_disasm(const std::vector<u1> &code, u4 len)
20{
21 // Redireciona stdout para um pipe temporário
22 fflush(stdout);
23 int saved = dup(STDOUT_FILENO);
24 int pipefd[2];
25 pipe(pipefd);
26 dup2(pipefd[1], STDOUT_FILENO);
27 close(pipefd[1]);
28
29 disassembleCode(code, len, empty_cls);
30 fflush(stdout);
31
32 dup2(saved, STDOUT_FILENO);
33 close(saved);
34
35 char buf[1024] = {};
36 ssize_t n = read(pipefd[0], buf, sizeof(buf) - 1);
37 close(pipefd[0]);
38 if (n > 0) buf[n] = '\0';
39 return std::string(buf);
40}
