14// =============================================================================
15// Heap — área de memória compartilhada do diagrama do professor
16//
17// Guarda tudo que é alocado em tempo de execução: instâncias de objetos
18// (opcode new) e arrays (newarray / anewarray). Cada alocação devolve uma
19// referência inteira (int32) que é o índice da célula nesta tabela.
20//
21// Convenção de referências (igual a Value::data.ref):
22// 0 → null
23// 1, 2, 3... → índices válidos de objetos/arrays
24//
25// Por isso o slot 0 é reservado e nunca é usado para um objeto real.
26// =============================================================================
27
28// -----------------------------------------------------------------------------
29// HeapObject — uma instância de classe (resultado de `new`)
30//
31// Os campos de instância são guardados por nome (string do constant pool),
32// cada um com seu Value. O class_info aponta para a classe que foi instanciada,
33// usado para resolver hierarquia, getfield/putfield e checagem de tipos.
34// -----------------------------------------------------------------------------
35struct HeapObject {
36 const class_info* cls = nullptr; // classe da instância
37 std::string class_name; // nome totalmente qualificado
38 std::unordered_map<std::string, Value> fields; // campos de instância
39};

41// -----------------------------------------------------------------------------
42// HeapArray — um array de uma dimensão (resultado de newarray / anewarray)
43//
44// Todos os tipos de elemento (int, long, float, double, referência...) são
45// guardados como Value. O campo element_type identifica o tipo dos elementos
46// para que os opcodes *aload / *astore saibam como interpretá-los.
47// -----------------------------------------------------------------------------
48struct HeapArray {
49 // Códigos de tipo do opcode newarray (JVMS Table 6.5.newarray-A).
50 enum ElementType : uint8_t {
51 T_BOOLEAN = 4,
52 T_CHAR = 5,
53 T_FLOAT = 6,
54 T_DOUBLE = 7,
55 T_BYTE = 8,
56 T_SHORT = 9,
57 T_INT = 10,
58 T_LONG = 11,
59 T_REFERENCE = 0, // arrays de objetos (anewarray)
60 };
61
62 u1 element_type = T_REFERENCE;
63 std::string element_class; // nome da classe (apenas para arrays de referência)
64 std::vector<Value> elements;
65};

67// -----------------------------------------------------------------------------
68// HeapCell — célula da tabela do heap: ou um objeto, ou um array.
69//
70// Usa unique_ptr para garantir que apenas uma das duas alternativas existe em
71// memória a cada momento — semântica de union sem os problemas de C++11 com
72// tipos não-triviais (std::vector / std::unordered_map não cabem em union bruta).
73// Em C++17 isso seria std::variant<HeapObject, HeapArray>.
74// -----------------------------------------------------------------------------
75struct HeapCell {
76 enum class Kind { OBJECT, ARRAY } kind = Kind::OBJECT;
77 std::unique_ptr<HeapObject> object; // não-nulo quando kind == OBJECT
78 std::unique_ptr<HeapArray> array; // não-nulo quando kind == ARRAY
79};

96 int32_t allocateObject(const class_info* cls, const std::string& class_name) {

99 cell.object.reset(new HeapObject());
100 cell.object->cls = cls;

106 // Cria um array de tipo primitivo (newarray) e devolve sua referência.
107 int32_t allocateArray(u1 element_type, int32_t length) {
108 if (length < 0)
109 throw std::runtime_error("NegativeArraySizeException");
110
111 HeapCell cell;
112 cell.kind = HeapCell::Kind::ARRAY;
113 cell.array.reset(new HeapArray());
114 cell.array->element_type = element_type;
115 cell.array->elements.assign(static_cast<size_t>(length), defaultFor(element_type));
116 cells_.push_back(std::move(cell));
