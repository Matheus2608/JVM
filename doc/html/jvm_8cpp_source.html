24void splitPath(const std::string& arg, std::string& classpath, std::string& class_name) {
25 std::string path = arg;
26
27 // Remove o sufixo .class, se presente.
28 const std::string ext = ".class";
29 if (path.size() >= ext.size() &&
30 path.compare(path.size() - ext.size(), ext.size(), ext) == 0)
31 path = path.substr(0, path.size() - ext.size());
32
33 // Separa diretório do nome da classe.
34 std::size_t slash = path.find_last_of('/');
35 if (slash == std::string::npos) {
36 classpath = ".";
37 class_name = path;
38 } else {
39 classpath = path.substr(0, slash);
40 class_name = path.substr(slash + 1);
41 }
42}
